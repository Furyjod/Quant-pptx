Strategy Logic

# Core Strategy Implementation
def detect_crossover(fast_ma, slow_ma):
    # Detect when fast SMA crosses above slow SMA
    above = fast_ma > slow_ma
    above_prev = above.shift(1)
    
    # Generate signals with no look-ahead bias
    long_signal = (above & ~above_prev).shift(1)
    exit_signal = (~above & above_prev).shift(1)
    
    return long_signal, exit_signal
